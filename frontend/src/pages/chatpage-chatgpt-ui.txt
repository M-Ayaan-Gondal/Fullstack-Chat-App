import { useState } from "react";

  MoreVertical,
  Search,
  Users,
  Menu,
  User,
  LogOut,
  Bot,
} from "lucide-react";
import { useAuthStore } from "../store/useAuthStore";

function ChatPage() {
  const [menuOpen, setMenuOpen] = useState(false);
  const [sidebarWidth, setSidebarWidth] = useState(280); // default width
  const [resizing, setResizing] = useState(false);

    { id: 1, name: "Ayaan", lastMessage: "last Message...", time: "9:39pm" },
    { id: 2, name: "arif", lastMessage: "last Message...", time: "9:39pm" },

  ];

  // mouse down event
  const startResizing = () => {
    setResizing(true);
    document.body.style.cursor = "col-resize";
  };

  // mouse move event
  const handleResizing = (e) => {
    if (resizing) {
      const newWidth = e.clientX - 56; // 56 = thin sidebar width
      if (newWidth > 200 && newWidth < 400) {
        setSidebarWidth(newWidth);
      }
    }
  };

  // mouse up event
  const stopResizing = () => {
    setResizing(false);
    document.body.style.cursor = "default";
  };

  // listeners for resize
  if (typeof window !== "undefined") {
    window.onmousemove = handleResizing;
    window.onmouseup = stopResizing;
  }

  const { logout } = useAuthStore();

  return (
    <div className="h-screen w-full flex">
      {/* LEFT THIN SIDEBAR */}

        >
          <LogOut className="w-5 h-5" />
        </button>
      </div>

      {/* LEFT CHAT SIDEBAR */}
      <div
        style={{ width: `${sidebarWidth}px` }}
        className="border-r bg-white flex flex-col relative"
      >
        {/* Header */}
        <div className="flex justify-between items-center px-4 py-3">

          </button>
        </div>

            <Search className="w-4 h-4 text-gray-500" />

              type="text"
              placeholder="Search"
              className="ml-2 bg-transparent outline-none w-full text-sm"
            />
          </div>
        </div>

        {/* Chat List */}
        <div className="flex-1 overflow-y-auto">
          {chats.map((chat) => (
            <div
              key={chat.id}
              className="flex items-center px-4 py-3 hover:bg-gray-50 cursor-pointer border-b"
            >
              <div className="w-10 h-10 rounded-full bg-gray-300 flex items-center justify-center">
                <span className="text-sm font-bold uppercase">
                  {chat.name[0]}
                </span>
              </div>
              <div className="ml-3 flex-1">
                <div className="flex justify-between">
                  <h3 className="font-medium">{chat.name}</h3>
                  <span className="text-xs text-gray-500">{chat.time}</span>
                </div>
                <p className="text-xs text-gray-500 truncate">
                  {chat.lastMessage}
                </p>
              </div>
            </div>
          ))}
        </div>

        {/* Resizer */}
        <div
          onMouseDown={startResizing}
          className="absolute right-0 top-0 h-full w-1 cursor-col-resize hover:bg-gray-300"
        />
      </div>

      {/* CHAT WINDOW */}
      <div className="flex-1 flex flex-col">
        {/* Chat Header */}
        <div className="flex justify-between items-center px-4 py-3 border-b bg-white">
          <div className="flex items-center">
            <div className="w-10 h-10 rounded-full bg-gray-300 flex items-center justify-center">
              A
